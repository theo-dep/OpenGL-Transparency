cmake_minimum_required(VERSION 3.16)

project(opengl-transparency VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

option(BuildBSP "If set, build a transparency rendering with a Binary Spatial Partionning tree sorting" OFF)

find_package(OpenGL REQUIRED)
find_package(GLEW REQUIRED)
find_package(FreeGLUT CONFIG REQUIRED)
find_package(glm CONFIG REQUIRED)
find_package(assimp CONFIG REQUIRED)
find_package(Freetype CONFIG REQUIRED)
find_package(CMakeRC CONFIG REQUIRED)

set(BINARY_DIR ${CMAKE_SOURCE_DIR}/${CMAKE_BUILD_TYPE}/bin)
add_subdirectory(media)

if(WIN32)
    add_definitions(-DNOMINMAX)
endif()
add_subdirectory(src)
